
import { NextRequest } from 'next/server'
import {
  fetchTrendingVideos,
  extractTrendingTopics,
  getFallbackTopics,
  isYouTubeConfigured,
} from '@/lib/youtube'

export async function GET(req: NextRequest) {
  const niche = req.nextUrl.searchParams.get('niche') ?? 'finance'

  if (!isYouTubeConfigured()) {
    return Response.json({
      success: true,
      data: getFallbackTopics(niche),
      source: 'fallback',
    })
  }

  try {
    const videos = await fetchTrendingVideos('US', 50)
    const topics = extractTrendingTopics(videos, niche)
    return Response.json({ success: true, data: topics, source: 'youtube_api' })
  } catch (error) {
    const message = error instanceof Error ? error.message : String(error)
    return Response.json(
      { success: false, error: message },
      { status: 500 }
    )
  }
}
